Peta Interaktif Samarinda

Peta ini menampilkan data kecamatan di Kota Samarinda.


                    var map = L.map('map').setView([-0.5022, 117.1536], 11);

                    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
                        attribution: '© OpenStreetMap contributors'
                    }).addTo(map);

                    // Data kecamatan (contoh, perlu disesuaikan dengan data sebenarnya)
                    var kecamatan = [
                        { name: "Samarinda Utara", lat: -0.4478, lon: 117.1751, data: 1000 },
                        { name: "Samarinda Ulu", lat: -0.4797, lon: 117.1535, data: 1500 },
                        { name: "Samarinda Ilir", lat: -0.5025, lon: 117.1419, data: 2000 },
                        { name: "Samarinda Seberang", lat: -0.5333, lon: 117.1833, data: 1200 },
                    ];

                    kecamatan.forEach(function (kec) {
                        var marker = L.marker([kec.lat, kec.lon]).addTo(map);
                        marker.bindPopup("" + kec.name + "
Data: " + kec.data);
                    });
